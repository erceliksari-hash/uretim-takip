import streamlit as st import pandas as pd from datetime import datetime # --- SAYFA AYARLARI --- st.set_page_config(page_title="Üretim Takip Pro v141", layout="wide") # --- CSS: MODERLEŞTİRME --- st.markdown(""" """, unsafe_allow_html=True) # --- ŞİFRE KORUMASI --- def giris_ekrani(): if "auth" not in st.session_state: st.session_state["auth"] = False if not st.session_state["auth"]: st.title("🔐 Üretim Portalı") sifre = st.text_input("Giriş Şifresini Giriniz", type="1234") if st.button("Sisteme Giriş Yap"): if sifre == "percin123": # <--- İstediğin şifreyi buraya yaz st.session_state["auth"] = True st.rerun() else: st.error("Hatalı Şifre!") return False return True if giris_ekrani(): # --- SESSION STATE (VERİ DEPOLAMA) --- if "liste" not in st.session_state: st.session_state["liste"] = [] if "kutuphane" not in st.session_state: st.session_state["kutuphane"] = {"PERÇİN": 0.550, "Örnek İş": 0.125} # --- YAN MENÜ (SIDEBAR) --- with st.sidebar: st.header("⚙️ Kontrol Paneli") sayfa = st.radio("Sayfa Değiştir", ["🏠 Üretim Ekranı", "🏷️ Artikel Kütüphanesi", "📜 Günlük Arşiv"]) st.divider() st.write(f"📅 Tarih: {datetime.now().strftime('%d.%m.%Y')}") # --- 🏠 ÜRETİM EKRANI --- if sayfa == "🏠 Üretim Ekranı": st.title("🚀 Üretim Takip") with st.container(): col1, col2 = st.columns(2) with col1: art_no = st.selectbox("Artikel Seç veya Yaz", list(st.session_state["kutuphane"].keys())) te_degeri = st.session_state["kutuphane"].get(art_no, 0.0) adet = st.number_input("Üretilen Adet", min_value=0, step=1) verim = st.number_input("Verim (Kutu boşsa 1 kabul edilir)", value=1.0, step=0.1) with col2: rust = st.number_input("Rüst (Dakika)", min_value=0, step=1) gmk = st.number_input("GMK / T53 (Dakika)", min_value=0, step=1) notlar = st.text_input("Not / Auftrag No") # Hesaplama (Verim mantığı korundu) hesaplanan_dk = (adet * te_degeri) / verim if verim != 0 else 0 st.metric("Hesaplanan Dakika", f"{hesaplanan_dk:.2f} dk") if st.button("➕ LİSTEYE EKLE"): yeni_is = { "Auftrag": notlar, "Artikel": art_no, "Adet": adet, "TE": te_degeri, "Verim": verim, "Net DK": round(hesaplanan_dk, 2), "Rüst": rust, "GMK": gmk, "Toplam": round(hesaplanan_dk + rust + gmk, 2) } st.session_state["liste"].append(yeni_is) st.success("İş başarıyla eklendi!") # --- TABLO --- if st.session_state["liste"]: st.subheader("📊 Günlük Liste") df = pd.DataFrame(st.session_state["liste"]) st.table(df) # Toplam Özeti toplam_dk = df["Toplam"].sum() hedef = st.number_input("Günlük Hedef (Dakika)", value=465) fark = hedef - toplam_dk c1, c2, c3 = st.columns(3) c1.metric("Toplam Üretim", f"{toplam_dk:.2f} dk") c2.metric("Kalan Hedef", f"{fark:.2f} dk", delta=-fark, delta_color="inverse") if st.button("🗑️ Listeyi Temizle"): st.session_state["liste"] = [] st.rerun() # --- 🏷️ ARTIKEL KÜTÜPHANESİ --- elif sayfa == "🏷️ Artikel Kütüphanesi": st.title("📖 Artikel Kütüphanesi") col1, col2 = st.columns(2) with col1: yeni_art = st.text_input("Yeni Artikel Adı").upper() yeni_te = st.number_input("TE Değeri (Dakika)", format="%.3f") if st.button("💾 Kütüphaneye Kaydet"): if yeni_art: st.session_state["kutuphane"][yeni_art] = yeni_te st.success(f"{yeni_art} başarıyla kaydedildi.") else: st.error("Lütfen bir isim girin!") with col2: st.subheader("Kayıtlı Artikeller") lib_df = pd.DataFrame(list(st.session_state["kutuphane"].items()), columns=["Artikel", "TE"]) st.dataframe(lib_df, use_container_width=True) # --- 📜 GÜNLÜK ARŞİV --- elif sayfa == "📜 Günlük Arşiv": st.title("📜 Geçmiş Kayıtlar") st.info("Bu sürümde veriler geçici bellektedir.
